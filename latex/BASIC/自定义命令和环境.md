```tex
\documentclass{ctexart}

\newcommand\PRC{Peple's Republic of \emph{China}}	% 定义命令PRC: Peple's Republic of \emph{China}
\newcommand\loves[2]{#1 喜欢 #2}	% 定义命令loves: #1 喜欢 #2. 这个命令有2个参数，分别用#1和#2指代；参数最多为9个
\newcommand\love[3][喜欢]{#2#1#3}	% 定义命令love: #2#1#3, 这个命令有三个参数,分别用#1,#2,#3指代，#1默认为‘喜欢’

\newenvironment{myabstract}[1][摘要]
{\small \begin{center}\bfseries #1 \end{center} \begin{quotation}}
{\end{quotation}}								
% 定义环境myabstract,一共有1个参数，参数1默认‘摘要’
% 该自定应环境就相当于: 
%			\begin{center}
%			\bfseries #1
%				\end{center}
%			\begin{quotation}
%				用户在自定义环境中输入的内容
%			\end{quotation}

\begin{document}
	\PRC
	
	\loves{猫儿}{小鱼}
	
	\love{猫儿}{小鱼}	% love命令有默认参数，这里没有用[]指明#1,#1用默认参数

	\love[特别喜欢]{猫儿}{小鱼}	% 这里#1为特别喜欢
	
	\begin{myabstract}
		这是一段自定义格式的摘要。
	\end{myabstract}

	\begin{myabstract}[我的摘要]
		这是一段自定义格式的摘要。
	\end{myabstract}
\end{document}

```
