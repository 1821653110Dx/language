```tex
\begin{itemize}
		\item c
\end{itemize}
\begin{itemize}[*]
		\item c

		\item c
\end{itemize}
```

$1. \quad 2. \quad 3.$
```tex
\begin{enumerate}
			\item a
			\item b
\end{enumerate}
```

$1.\quad1.1\quad1.1.1$
```tex
\begin{enumerate}
	\item a
		\begin{enumerate}[label*=\arabic*.]
			\item a
				\begin{enumerate}[label*=\arabic*.]
					\item a
						\begin{enumerate}[label*=\arabic*.]
								\item a
								\item b
						\end{enumerate}	
					\item b
				\end{enumerate}	
			\item b
		\end{enumerate}
	\item b
\end{enumerate}
```	

$(1) (2) (3)$
```tex
\begin{enumerate}[label = (\arabic*)]
	\item a
	\item b
\end{enumerate}
```

$1) 2) 3)$
```tex
\begin{enumerate}[label = \arabic*)]
	\item a
	\item b
\end{enumerate}
```

$一、 二、$
```tex

	\item a
	\item b
\end{enumerate}
```
