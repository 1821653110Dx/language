```python
import numpy as np
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D


"def arg: X, Y"
X = np.arange(-4,4,0.25)    # create a sequence : [-4, 4), d = 0.25 | save to X
Y = np.arange(-4,4,0.25)    # create a sqeuentce : [-4, 4), d = 0.25 | save to Y

"def func : Z(X, Y)"
X,Y = np.meshgrid(X, Y) # convert X, Y into 2D arrays

Z = np.sin(np.sqrt(X**2 + Y**2))

"generate 3Dplots in figure1: Z(X, Y)"
fig1 = plt.figure() # create figure1 | save to fig1
ax1 = Axes3D(fig1)  # create 3D axes for fig1 | save to ax1
fig1.add_axes(ax1) # add ax1 to fig1

ax1.plot_surface(X, Y, Z, rstride=5, cstride=5, cmap=plt.get_cmap('rainbow'))  # generate Z(X,Y) in ax1, set mesh : [len_x, len_y, cmap] = [5,5,rainbow]   # cmap = colormap, len_x = length along x_axis

"show figures"
plt.show()
```
